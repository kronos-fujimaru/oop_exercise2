#### Point.java
```java
public class Point {
    int x;
    int y;

    Point(int x, int y) {
        this.x = x;
        this.y = y;
    }

    int getArea() {
        return x * y;
    }

    int getArea(int x, int y) {
        int width = 0;
        int height = 0;

        if (x - this.x >= 0) {
            width = x - this.x;
        } else {
            width = this.x - x;
        }

        if (y - this.y >= 0) {
            height = y - this.y;
        } else {
            height = this.y - y;
        }

        return width * height;
    }
}
```
