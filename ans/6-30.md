import java.util.ArrayList;
public class MobileItemShop {
    ArrayList<PortableTelephone> items = new ArrayList<PortableTelephone>();

    int purchase(PortableTelephone portableTelephone) {
        items.add(portableTelephone);
        return 10000;
    }
}
