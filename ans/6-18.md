public class Enshu18 {
    int number = 50;
    
    void even() {
        display();
        number /= 2;
    }
    
    void odd() {
        display();
        number = number * 3 + 1;
    }
    
    int getNumber() {
        return number;
    }
    
    void display() {
        System.out.println(number);
    }
    
    public static void main(String[] args) {
        Enshu18 enshu = new Enshu18();
        while (enshu.getNumber() != 1) {
            if (enshu.getNumber() % 2 == 0) {
                enshu.even();
            } else {
                enshu.odd();
            }
        }
        enshu.display();
    }
}
