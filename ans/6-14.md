class Enshu14 {
    static boolean triangleCheck(int a, int b, int c) {
        if (a >= b && a >= c) {
            return a < b + c;
        } else if (b >= a && b >= c) {
            return b < a + c;
        } else {
            return c < a + b;
        }
    }
    
    public static void main(String[] args) {
        if (Enshu14.triangleCheck(3, 4, 5)) {
            System.out.println("三角形として成立します");
        } else {
            System.out.println("三角形として成立しません");
        }
    }
}
