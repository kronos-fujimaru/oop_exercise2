class Enshu13 {
    int sum;
    
    void add(int sum) {
        this.sum += sum;
    }
    
    int get() {
        return sum;
    }
    
    public static void main(String[] args) {
        Enshu13 enshu = new Enshu13();
        enshu.add(3);
        enshu.add(2);
        enshu.add(6);
        System.out.println(enshu.get());
    }
}
