public class Enshu17 {
    int number = 50;
    
    void even() {
        System.out.println(number);
        number /= 2;
    }
    
    void odd() {
        System.out.println(number);
        number = number * 3 + 1;
    }
    
    int getNumber() {
        return number;
    }
    
    public static void main(String[] args) {
        Enshu17 enshu = new Enshu17();
        while (enshu.getNumber() != 1) {
            if (enshu.getNumber() % 2 == 0) {
                enshu.even();
            } else {
                enshu.odd();
            }
        }
        System.out.println(enshu.getNumber());
    }
}
