class Enshu12 {
    int sum;
    
    void add(int a) {
        sum += a;
    }
    
    int get() {
        return sum;
    }
    
    public static void main(String[] args) {
        Enshu12 enshu = new Enshu12();
        enshu.add(3);
        enshu.add(2);
        enshu.add(6);
        System.out.println(enshu.get());
    }
}
